📄 Configuration (.env file)

# Create a file named .env in the videogen/ folder

# Flask settings
SECRET_KEY=any-random-string-abc123
PORT=5000

# FREE - Pexels B-Roll (pexels.com/api)
PEXELS_API_KEY=your-free-pexels-key

# FREE - Pixabay B-Roll + Music (pixabay.com/api/docs)
PIXABAY_API_KEY=your-free-pixabay-key

# FREE - Freesound SFX (freesound.org/apiv2/apply)
FREESOUND_API_KEY=your-free-freesound-key

# Whisper model size (tiny/base/small/medium)
WHISPER_MODEL=base
